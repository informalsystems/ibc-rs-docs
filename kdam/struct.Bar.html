Expand description

Core implemention of console progress bar.

§Example

use kdam::{tqdm, Bar, BarExt};

let mut pb = Bar::new(100);
// let mut pb = tqdm!(total = 100);
// let mut pb = Bar::builder().total(100).build().unwrap();

for _ in 0..100 {
    pb.update(1).unwrap();
}
